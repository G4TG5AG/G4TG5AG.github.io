  • C#でショートカットファイルを作成しよう！
    • このサイトでは、C#でショートカットファイルを作成する方法を紹介します。

      C#で、インストーラーなどを作る時などに、「デスクトップにショートカットを作りたいな～」と思うことはありませんか？

      今回は、ショートカットを作成する方法を紹介します！

      メソッドとプロパティで作成できるようにコードを作成しましたので、簡単にできると思います！

      前提コードはこちらです。

      ソースコードを表示 ソースコードを非表示 
      namespace G4TG5AG.Shortcut
{
    /**
    <summary>ショートカットを作成するためのメソッドおよびプロパティを提供します。</summary>
    */
    internal class Shortcut
    {
        /**
        <summary>ショートカットファイルを作成する場所を指定します。</summary>
        */
        public string FileName
        {
            get;
            set;
        }

        /**
        <summary>ショートカットのリンク先を指定します。</summary>
        */
        public string TargetPath
        {
            get;
            set;
        }

        /**
        <summary>ショートカットのリンク先を開く際のコマンドライン引数を指定します。</summary>
        */
        public string Arguments
        {
            get;
            set;
        }

        /**
        <summary>ショートカットのリンク先の作業フォルダを指定します。</summary>
        */
        public string WorkingDirectory
        {
            get;
            set;
        }

        /**
        <summary>ショートカットファイルのアイコンを指定します。</summary>
        */
        public string IconLocation
        {
            get;
            set;
        }

        /**
        <summary>ショートカットファイルのコメントを指定します。</summary>
        */
        public string Description
        {
            get;
            set;
        }

        /**
        <summary>ショートカットのリンク先を開く際のウィンドウの表示方法を指定します。</summary>
        */
        public enum WindowStyles
        {
            Normal = 1,
            Maximized = 3,
            Minimized = 7
        }

        /**
        <summary><see cref="Shortcut.WindowStyles"/> を使用して、ショートカットのリンク先を開く際のウィンドウの表示方法を指定します。</summary>
        */
        public WindowStyles WindowStyle
        {
            get;
            set;
        } = WindowStyles.Normal;

        /**
        <summary><see cref="Shortcut"/> クラスのプロパティを使用して、ショートカットを作成します。</summary>
        */
        public void ShortcutCreate()
        {
            if (System.IO.Directory.Exists(System.IO.Path.GetDirectoryName(FileName)) == true)
            {
                if (FileName.Substring(FileName.Length - 4, 4).ToLower() == ".lnk" || FileName.Substring(FileName.Length - 4, 4).ToLower() == ".url")
                {
                    var WshShellType = System.Type.GetTypeFromCLSID(new System.Guid("72C24DD5-D70A-438B-8A42-98424B88AFB8"));
                    dynamic WshShell = System.Activator.CreateInstance(WshShellType);
                    var shortcut = WshShell.CreateShortcut(FileName);
                    shortcut.TargetPath = TargetPath;
                    shortcut.Arguments = Arguments;
                    if (WorkingDirectory == string.Empty)
                    {
                        shortcut.WorkingDirectory = System.IO.Path.GetDirectoryName(TargetPath);
                    }
                    else
                    {
                        shortcut.WorkingDirectory = WorkingDirectory;
                    }
                    if (IconLocation == string.Empty)
                    {
                        shortcut.IconLocation = $@"{TargetPath},0";
                    }
                    else
                    {
                        try
                        {
                            shortcut.IconLocation = IconLocation;
                        }
                        catch (System.Exception)
                        {

                        }
                    }
                    shortcut.Description = Description;
                    shortcut.WindowStyle = WindowStyle;
                    shortcut.Save();
                    System.Runtime.InteropServices.Marshal.FinalReleaseComObject(shortcut);
                    System.Runtime.InteropServices.Marshal.FinalReleaseComObject(WshShell);
                }
                else
                {
                    throw new System.ArgumentException("ショートカットのパス名は .lnk または .url で終わらなければなりません。");
                }
            }
            else
            {
                throw new System.IO.DirectoryNotFoundException();
            }
        }
    }
}

      このコードを使うと、Shortcutクラスを作成して、プロパティを変更してからメソッドを実行するだけでショートカットを作成できます！

      コード例はこちらです。

      ソースコードを表示 ソースコードを非表示 
      //usingディレクティブが記述されていることを前提としています。
Shortcut shortcut = new Shortcut(); //Shortcutクラスを作成
shortcut.FileName = @"C:\hogehoge.lnk"; //ショートカットの保存場所を指定
shortcut.TargetPath = @"C:\hogehoge\hogehoge.txt"; //ショートカットのリンク先を指定
shortcut.Arguments = "/hogehoge"; //コマンドライン引数を設定
shortcut.Description = "hogehoge.txtへのリンクです。"; //ショートカットの説明を設定
shortcut.WindowStyle = Shortcut.WindowStyles.Maximized; //実行時のウィンドウを指定
shortcut.IconLocation = @"C:\hogehoge\hogehoge.ico"; //アイコンを指定
shortcut.ShortcutCreate(); //ショートカットを作成
